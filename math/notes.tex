\section{Basic Mathematics}

\Problem If $$1^2x_1 + 2^2 x_2 + 3^2 x_3 + 4^2 x_4 + ... + 9^2 x_9 = 1$$
$$2^2x_1 + 3^2 x_2 + 4^2 x_3 + 4^2 x_4 + ... + 10^2 x_9 = 11$$
$$1^2x_1 + 3^2 x_2 + 5^2 x_3 + 7^2 x_4 + ... + 17^2 x_9 = 111$$
then find:
\begin{itemize}
  \item[a.)] $3^2x_1 + 4^2 x_2 + 5^2 x_3 + 6^2 x_4 + ... + 11^2 x_9 $
  \item[b.)] $3^2x_1 + 5^2 x_2 + 7^2 x_3 + 9^2 x_4 + ... + 19^2 x_9 $
\end{itemize}

\TheSolution \be \sum\limits_{r=1} ^9 (r^2 x_r) = 1 \ee
\be \sum\limits_{r=1} ^9 ((r+1)^2 x_r)  = 11 \ee
\be \sum\limits_{r=1} ^9 ((2r-1)^2x_r) = 111 \ee
a.) let \be \sum\limits_{r=1} ^9 ((r+2)^2x_r) = \sum\limits_{r=1} ^9 ((Xr^2 +Y(r+1)^2 + Z(2r-1)^2 ) x_r) \ee
\be \sum\limits_{r=1} ^9 ((r^2+4+4r)x_r)=\sum\limits_{r=1} ^9 ((r^2(X+Y+4Z)+r(2Y-4Z)+(Y+Z))x_r) \ee
so,
\be X+Y+Z=1 \ee
\be 2Y-4Z=4 \ee
\be Y+Z=4 \ee
so,
\be X=-5 \ee
\be Y=\frac{10}{3} \ee 
\be Z=\frac{2}{3} \ee
so,
\be \sum\limits_{r=1} ^9 ((r+2)^2x_r) = \sum\limits_{r=1} ^9 (((-5)r^2 +(\frac{10}{3})(r+1)^2 + (\frac{2}{3})(2r-1)^2 ) x_r) \ee
\be \sum\limits_{r=1} ^9 ((r+2)^2x_r)\  =\  (-5) \sum\limits_{r=1} ^9 (r^2 x_r) \ +\  \frac{10}{3} \sum\limits_{r=1} ^9 ((r+1)^2 x_r)\ +\  \frac{2}{3}\sum\limits_{r=1} ^9 ((2r-1)^2x_r) \ee
so by (1.1), (1.2), and (1.3),
\be \sum\limits_{r=1} ^9 ((r+2)^2x_r) = (-5)(1) + \frac{10}{3}(11) +  \frac{2}{3}(111) \ee 
\be \sum\limits_{r=1} ^9 ((r+2)^2x_r) = \frac{317}{3} \ee
